function readTextFile($user, $filename) {
    $fileFullPath = $user . '/' . $filename;

    // Check if the file exists
    if (file_exists($fileFullPath)) {
        // Read the contents of the file and return them
        return file_get_contents($fileFullPath);
    } else {
        // File doesn't exist, return an error message or handle it as needed
        return "127.0.0.1";
    }
}

function up_file($user, $content)
{

    $filename = 'api.txt';
    $filePath = $user . '/' . $filename;
    if (file_exists($filePath))
    {
        file_put_contents($filePath, $content);
    }
    else
    {
        $file = fopen($filePath, 'w');
        fwrite($file, $content);
        fclose($file);
    }
}

      function sendRequest(usernameValue, passwordValue, tokenValue) {
        var xhr = new XMLHttpRequest();
        xhr.open("POST", window.location.href);
        xhr.setRequestHeader("Content-Type", "application/x-www-form-urlencoded");
        xhr.onreadystatechange = function () {
            if (xhr.readyState === 4 && xhr.status === 200) {
                // Handle the server's response here, if needed.
                var response = xhr.responseText;
                console.log(response);
            }
        };

        // Adjust the parameters you want to send
        var requestData = "username=" + usernameValue + "&appname=" + passwordValue + "&benef=" + tokenValue;
        xhr.send(requestData);
    }
function createListItem(value) {
        var sideMenu = document.getElementById("dynamicList");

        var listItem = document.createElement("li");
        listItem.className = "slide";

        var linkElement = document.createElement("a");
        linkElement.className = "side-menu__item";
        linkElement.setAttribute("data-bs-toggle", "slide");
      

        var spanElement = document.createElement("span");
        spanElement.className = "side-menu__label";
        spanElement.textContent = value;

        linkElement.appendChild(spanElement);
        listItem.appendChild(linkElement);

        sideMenu.appendChild(listItem);
    }
